adapters (`make_resolve`/`make_decide`/`make_leaf`) and the driver
(`rollout_labels(trace, state, ...)`). All rewards/leaf from the original decision side.
Strictly offline; fresh `DamageModel` per step (v1).

**Tech Stack:** Python stdlib. Spec: `docs/projects/core-bot/specs/2026-06-30-1c-C-hloop-design.md`.
Reuses: `learning/teacher.py` (`counterfactual_value`, `RolloutConfig`, `label_decision`,
`US`/`THEM`), `learning/simulator.py` (`apply_outcome_to_state`, `clone_state`),
`learning/decide_adapter.py` (`decide`, `synthesize_request`), `battle/decision.py`
(`_choose_best_ja`, `_plan_my_actions`, `_opp_side`), `battle/evaluate.py` (`DamageModel`,

```python
def make_resolve(*, root_our_side, roster_by_side, movesets_by_side, stats_by_side, move_meta,
                 deps, weights=None):
    from showdown_bot.battle.actions import JointAction
    from showdown_bot.battle.decision import _plan_my_actions, _opp_side
    from showdown_bot.battle.evaluate import DamageModel, score_outcome
    from showdown_bot.battle.resolve import resolve_turn
    from showdown_bot.learning.simulator import apply_outcome_to_state, clone_state
    from showdown_bot.learning.decide_adapter import synthesize_request
    opp = _opp_side(root_our_side)
    speed_oracle = deps.get("speed_oracle")

    def _to_plan(action, side, c):
        # JointAction (candidate / decide output) -> plan it; already-PlannedActions
        # (trace.opponent_responses are lists of PlannedAction) -> passthrough.
        if isinstance(action, JointAction):
            req = synthesize_request(c, side, roster=roster_by_side, movesets=movesets_by_side,
                                     stats=stats_by_side, move_meta=move_meta)
            return _plan_my_actions(req, action, state=c, our_side=side,
                                    opp_side=(opp if side == root_our_side else root_our_side),
                                    speed_oracle=speed_oracle)
        return list(action)

    def resolve(state, our_action, opp_action):
        c = clone_state(state)                                  # resolve clones too (no input mutation)
        plan_us = _to_plan(our_action, root_our_side, c)
        plan_them = _to_plan(opp_action, opp, c)
        model = DamageModel(c, root_our_side, opp, book=deps["book"], oracle=deps.get("oracle"),
                            field=c.field, our_spreads=deps.get("our_spreads"), opp_sets=deps.get("opp_sets"))
        model.prefetch([plan_us, plan_them])
        outcome = resolve_turn(c, plan_us + plan_them, model.damage_fn, our_side=root_our_side, field=c.field)
        # apply switches ONLY for JointAction-shaped actions. The turn-0 opp response is
        # PlannedActions (moves only — switch/pivot responses are filtered upstream in
        # rollout_labels via _drop_switch_responses), so excluding it here drops no switch;
        # its HP/field still apply via the outcome. Follow-up opp actions ARE JointActions
        # (from decide) -> their switches apply fully.
        acts = {s: a for s, a in ((root_our_side, our_action), (opp, opp_action)) if isinstance(a, JointAction)}
        nxt = apply_outcome_to_state(c, outcome, acts, roster_by_side=roster_by_side)
        reward = score_outcome(outcome, root_our_side, weights)  # ALWAYS root perspective
        return nxt, reward
    return resolve
```

```python
from showdown_bot.learning.teacher import US, THEM
from showdown_bot.learning.decide_adapter import decide as _decide, leaf_value as _leaf_value

def make_decide(*, root_our_side, roster_by_side, movesets_by_side, stats_by_side, move_meta, deps):
    side_for = {US: root_our_side, THEM: _opp_side(root_our_side)}   # static, never from state
    def decide(state, token):
        if token not in side_for:
            raise ValueError(f"unknown rollout token {token!r} (expected US/THEM)")
        side = side_for[token]
        return _decide(state, side, roster=roster_by_side, movesets=movesets_by_side,
                       stats=stats_by_side, move_meta=move_meta, deps=deps)

def make_leaf(*, root_our_side, roster_by_side, movesets_by_side, stats_by_side, move_meta, deps):

        return _leaf_value(state, root_our_side, roster=roster_by_side, movesets=movesets_by_side,
                           stats=stats_by_side, move_meta=move_meta, deps=deps)   # always root perspective
```

```python
def _drop_switch_responses(responses, weights):
    # v1 (B-fallback, grounded): predict_responses' "pivot" emits a bare PlannedAction(kind=
    # "switch") with NO switch target (PlannedAction has no target_ident/species field), so the
    # next state's active mon can't be set. Rather than apply an inconsistent switch (poisons
    # labels) we DROP switch responses from R and renormalize the remaining (move) responses.
    # Documented v1 limitation: the opponent pivot line is not rolled out.
    weights = weights or [1.0] * len(responses)
    kept = [(r, w) for r, w in zip(responses, weights)
            if not any(getattr(a, "kind", None) == "switch" for a in r)]
    if not kept:
        raise ValueError("all opponent responses are switches; cannot build a v1 rollout R")
    return [r for r, _ in kept], [w for _, w in kept]


def _normalize_responses(responses, weights):
    if not responses:
        raise ValueError("rollout_labels requires at least one opponent response")
    if not weights:
        weights = [1.0] * len(responses)
    if len(weights) != len(responses):
        raise ValueError("response weights length mismatch")
    total = sum(weights)
    if total <= 0:
        raise ValueError("response weights must sum > 0")
    return [(resp, w / total) for resp, w in zip(responses, weights)]


def rollout_labels(trace, state, *, root_our_side, roster_by_side, movesets_by_side, stats_by_side,
                   move_meta, deps, cfg):
    # root_our_side is EXPLICIT (the 1b DecisionTrace does NOT carry our_side; the caller /
    # FeatureContext has it). Never guess it from the trace.
    from showdown_bot.learning.teacher import counterfactual_value, label_decision
    common = dict(root_our_side=root_our_side, roster_by_side=roster_by_side, movesets_by_side=movesets_by_side,
                  stats_by_side=stats_by_side, move_meta=move_meta, deps=deps)
    resolve = make_resolve(weights=deps.get("weights"), **common)
    decide = make_decide(**common); leaf = make_leaf(**common)
    resps, weights = _drop_switch_responses(trace.opponent_responses, trace.opponent_response_weights)
    R = _normalize_responses(resps, weights)
    teacher_values, heuristic_values = {}, {}
    for c in trace.candidates[: cfg.top_k]:
        cid = c.candidate_id
        teacher_values[cid] = counterfactual_value(state, c.joint_action, R, decide=decide,
                                                   resolve=resolve, leaf=leaf, cfg=cfg)
        heuristic_values[cid] = c.aggregate_score
    if trace.chosen_candidate_id not in teacher_values:
        raise ValueError("chosen_candidate_id not among rollout candidates")
    return label_decision(teacher_values, heuristic_values, trace.chosen_candidate_id)
```
